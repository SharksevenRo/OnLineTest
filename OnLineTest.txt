Create database OnlineTest;
use OnlineTest;

create table User(userId nvarchar(50) not null primary key,userName nvarchar(20),password nvarchar(50),type nvarchar(10));
create table Question(questionId  nvarchar(50) not null primary key,content text,answer nvarcahr(255),type nvarchar(10));
create table Record(recordId  nvarchar(50) not null primary key,userId nvarchar(50),mark nvarchar(255),foreign key(userId) references User(userId),time timestamp);
create table Message(messageId nvarchar(50) not null primary key,content text,userName nvarchar(20),teacherId nvarchar(50), foreign key(teacherId) references User(userId),time timestamp)
