use OnlineTest;

create table Question(questionId  nvarchar(50) not null primary key,userId nvarchar(50),content text,answer nvarcahr(255),type nvarchar(10),foreign key(userId) references user(userId));

create table Message(messageId nvarchar(50) not null primary key,content text,toUser nvarchar(20),fromUser nvarchar(20),time timestamp)
